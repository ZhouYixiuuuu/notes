```c++
#include <iostream>
#include <algorithm>
#include <cstring>
#include <cmath>

using namespace std;
typedef long long LL;
const LL Mod = 998244353;

LL qmi(LL a, LL k)
{
    LL res = 1;
    while (k)
    {
        if (k & 1) res = res * a % Mod;
        a = a * a % Mod;
        k >>= 1;
    }
    return res;
}

int main()
{
    LL n, m;
    cin >> n >> m;
    m += n;
    LL ans = 1;
    while (n < m)
    {
        LL a = log2(n + 1);
        //cout << n + 1 << ' ' << a << endl;
        LL next = min(m, (LL)pow(2, a + 1) - 1);
        LL temp = (1 - 1 * qmi((LL)pow(2, a), Mod - 2) + Mod) % Mod;
        ans = ans * qmi(temp, next - n) % Mod;
        n = next;
    }
    cout << ans << endl;
    return 0;
}
```
